\section*{Reinforcement Learning}

Our formulation of the problem based on \cite{sutton_barto_2018} is 
to find $\pi: \mathcal{S} \to \mathcal{P}(\mathcal{A})$ ($\mathcal{P}$ is a p-measure) such that  
%
\begin{align*}
    \max_{\pi} J(\pi) &= \max_{\pi} \mathsf{E}_{\tau \sim p(\tau|\pi)} 
    \left[ 
        \sum_{i=0}^{\infty} \gamma^i \underbrace{R(\bm{s}_i, a_i, \bm{s}_{i+1})}_{r_i}
    \right]
    \\
    p(\tau \mid \pi) &= p(\bm{s}_0) \cdot \prod_{t=0}^{\infty} \pi(a_t|\bm{s}_t) \cdot P_{a_t}(\bm{s}_{t+1}, \bm{s}_t)
    \\
    \bm{s}_{t+1} &\sim P_{a_t}(\cdot , \bm{s}_t) 
    \quad
    a_t \sim \pi(\cdot | \bm{s}_t)
\end{align*}

Where $\tau$ is a trajectory of the system, which consists of the states, actions and rewards at each time step.
%
\small{
    \begin{align*}
        \bm{\tau}_t
        &=
        \left\{
            \vphantom{\int}
            \bm{s}_{t+i}, a_{t+i}, r_{t+i}, \bm{s}_{t+i+1} 
        \right\}_{i=0}^{\infty}
        \Rightarrow
        J(\rot{\pi}) 
        =
        \mathsf{E}
        \left[
            \vphantom{\int}
            R(\bm{\tau}_0)
        \right]
    \end{align*}
}

\subsection*{Policy Gradient Methods}

Let $\pi_{\theta}$ be a parameterized policy with $\theta \in \mathbb{R}^d$.
    %
\begin{align*}
    \nabla_{\theta} J(\theta)
    &=
    \nabla_{\theta} \mathsf{E}_{\tau \sim p(\tau|\theta)}
    \left[
        R(\bm{\tau})
    \right]
    \\
    &= 
    \int_{\mathcal{T}} R(\bm{\tau}) \cdot 
    \underbrace{
        \nabla_{\theta} \log(p(\bm{\tau}\mid \theta)) \cdot p(\bm{\tau}\mid \theta)
    }_{\nabla_{\theta} p(\bm{\tau}|\theta)}
    \,\mathrm{d}\bm{\tau}
    \\
    &=
    \mathsf{E}_{\tau \sim p(\tau|\theta)}
    \left[
        R(\bm{\tau}) \cdot \nabla_{\theta} \log(p(\bm{\tau}\mid \theta))
    \right]
\end{align*}

The Logarithm applied to the distribution $p(\bm{\tau} \mid \theta)$:
%
\begin{align*}
    \nabla_{\theta}
    &
    \log
    \left[
        p(\bm{s}_0) \cdot \prod_{k=0}^{T} \pi_{\theta}(a_k|\bm{s}_k) \cdot P_{a_k}(\bm{s}_{k+1}, \bm{s}_k)
    \right]
    \\
    &=
    \sum_{k=0}^{T} \nabla_{\theta} 
    \log\left[
        \pi_{\theta}(a_k|\bm{s}_k)
    \right]
\end{align*}

Let $R(\bm{\tau}) = \sum_{i=0}^{T-1} r_i + r_T$ where $r_i = R(\bm{s}_i, a_i, \bm{s}_{i+1})$
%
\begin{align*}
    &\mathsf{E}_{\tau \sim p(\tau|\theta)}
    \left[
        R(\bm{\tau}) \cdot \sum_{k=0}^{T} \nabla_{\theta} 
        \log\left[
            \pi_{\theta}(a_k|\bm{s}_k)
        \right]
    \right]
    \\
    &=
    \sum_{i=0}^{T}
    \sum_{k=0}^{T}
    \mathsf{E}_{\tau}
    \left[
        r_i 
        \cdot 
        \nabla_{\theta} \log
        \left[
            \pi_{\theta}(a_k|\bm{s}_k)
        \right]
    \right]
\end{align*}

For fixed $i<k$ the law of total expectation yields:
%
%
\begin{align*}
    &\mathsf{E}_{\tau \sim p(\tau|\theta)}
    \left[
        r_i 
        \cdot 
        \nabla_{\theta} 
        \log
        \left[
            \pi_{\theta}(a_k|\bm{s}_k)
        \right]
    \right]
    \\
    &=
    \mathsf{E}_{a_k, \bm{s}_k}
    \left[
        \nabla_{\theta} \log
        \left[
            \pi_{\theta}(a_k|\bm{s}_k)
        \right]
        \cdot
        \underbrace{
            \mathsf{E}_{\tau}
            \left[
                r_i 
                \mid a_k, \bm{s}_k
            \right]
        }_{
            \mathsf{E}_{\tau}[r_i \mid \bm{s}_k] = g(\bm{s}_k)
        }
    \right]
    \\
    &=
    \mathsf{E}_{\bm{s}_k}
    \left[
        \mathsf{E}_{a_k \mid \bm{s}_k}
        \left[
            \vphantom{\int}
            \left[ 
                g(\bm{s}_k)
                \cdot 
                \nabla_{\theta} \log\left( \pi_{\theta}(a_k|\bm{s}_k) \right) 
            \right]
            \mid \bm{s}_k
        \right]
    \right]
\end{align*}
%
Now using $g(\bm{s}_k)$ is independent of $a_k$ and therefore
%
\small{
    \begin{align*}
        &
        \mathsf{E}_{\bm{s}_k}
        \left[
            g(\bm{s}_k)
            \mathsf{E}_{a_k \mid \bm{s}_k}
            \left[
                \left[ 
                    \nabla_{\theta} \log\left( \pi_{\theta}(a_k|\bm{s}_k) \right) 
                \right]
                \mid \bm{s}_k
            \right]
        \right]
        \\
        &=
        \mathsf{E}_{\bm{s}_k}
        \left[
            g(\bm{s}_k)
            \cdot
            \int_{\mathcal{A}} 
            \underbrace{
                \nabla_{\theta} 
                \log
                \left( 
                    \pi_{\theta}( a |\bm{s}_k) 
                \right) 
                \cdot 
                \pi_{\theta}(a |\bm{s}_k) 
            }_{
                \nabla_{\theta} \pi_{\theta}(a|\bm{s}_k)
            }
            \mathrm{d}a
        \right]
        \\
        &=
        \nabla_{\theta} 
        \underbrace{
            \int_{\mathcal{A}} \pi_{\theta}(a|\bm{s}_k) \mathrm{d}a
        }_{
            =1
        }
        = 0
    \end{align*}
}

We conclude that for $i<k$ the expectation is zero and 
we denote $R_t = \sum_{k=t}^{T} r_k$ 
and obtain the formula:
%
\begin{align*}
    \nabla_{\theta} J(\theta) 
    &=
    \sum_{t=0}^{T} 
    \mathsf{E}_{\tau \sim p(\tau|\theta)}
    \left[
        \nabla_{\theta} 
        \log
        \left( 
            \pi_{\theta}(a_t|\bm{s}_t) 
        \right)
        \cdot
        R_t
    \right]
\end{align*}

Conditioned on $(\bm{s}_t,a_t)$ we get for $\nabla_{\theta} J(\theta)$:
%
\begin{align*}
    &\nabla _{\theta} J(\theta)
    =
    \mathsf{E}
    \left[
        \vphantom{\int}
        \mathsf{E}
        \left[
            \nabla_{\theta} 
            \log
            \left( 
                \pi_{\theta}(a_t|\bm{s}_t) 
            \right)
            \cdot
            R_t
            \mid a_t, \bm{s}_t
        \right]
    \right]
    \\
    &=
    \sum_{t=0}^{T}
    \mathsf{E}_{a_t, \bm{s}_t}
    \left[
        \nabla_{\theta} 
        \log
        \left( 
            \pi_{\theta}(a_t|\bm{s}_t) 
        \right)
        \underbrace{
            \mathsf{E}_{\tau \sim p(\tau|\theta)}
            \left[
                R_t \mid a_t, \bm{s}_t
            \right]
            }_{
                Q^{\pi_{\theta}}(\bm{s}_t, a_t)
            }
    \right]
\end{align*}

\subsection*{Application on Control Problems}

We parametrize the control $u(x,t) = u(t,x \mid \theta)$ and 
relax the problem by considering 
$a_t \sim \pi(u(\bm{s}_t, \theta))$. 

\begin{tabular}{ l| l}
    Control Problem   & RL Problem \\
    \hline 
    $x(t)$                
    & 
    $\bm{s}_t = 
    \begin{bmatrix}
        t \\
        x(t)
    \end{bmatrix}$
    \\
    $u(t)$                
    & 
    $a_t \sim \mathcal{N}(u(\bm{s}_t, \theta), \sigma)$
    \\
    $\displaystyle \int_0^T L(x,u) \dt + \Phi(x(T))$                
    & 
    $\displaystyle \mathsf{E}_{\tau}\left[ \sum_{i=0}^{N} r_i \right]$
    \\
    $\displaystyle x'(t) = f(x,u)$
    &   
    $\displaystyle \bm{s}_{t+1} \sim P_{a_t}(\cdot,\bm{s}_{t})$
\end{tabular}

% Schemata Policy Gradient Methode
% \mathcal{U} = \{u:[0,T] \to \mathbb{R} \mid u \text{ measurable} \}

\begin{itemize}
    \item[1.] Let $\mathcal{U} = \{u:[0,T] \to \mathbb{R} \mid u \text{ measurable} \}$
    %
    \begin{align*}
            \min_{u \in \mathcal{U}} & \int_0^T
            L(X_t,u) \dt 
            + \Phi(X_T),  
            \quad
            \\
            \dX_t &= f(X_t,u(t)) \dt + \sigma \dW_t, 
            \quad X_0= x_0
    \end{align*}
    \item[2.] Relaxation of the Problem: $\mathcal{U} \subset \mathcal{U}_{stoch}$
        %
    \begin{align*}
        \mathcal{U}_{stoch} &= \{ \pi: \mathcal{S} \to \mathcal{P}(\mathbb{R}) 
        \mid \pi(\cdot | \bm{s}_t) \text{ is p-measure} \}
        \\
        u: & \mathbb{R}^{n_x} \times [0,T] \to \mathbb{R} \Rightarrow 
        \pi( \cdot | \bm{s}_t) = \delta_{u(\bm{s}_t)}
    \end{align*}

    \item[3.] Discretize: $\bm{a} = \left\{ a_i \right\}_{i=0}^{N-1}$ $a_i \sim \pi(u(\bm{s}_i, \theta))$ 
    %
    \small{
        \begin{align*}
        \log\left[
            \underbrace{
                \prod_{i=0}^{N-1} \pi_{\theta}(a_i | \bm{s}_i) 
            }_{
                \pi_{\theta}(\bm{a})
            }
        \right] 
        = 
        \sum_{i=0}^{N-1} \log\left[\pi_{\theta}(a_i | \bm{s}_i)\right]
    \end{align*}
    }
    \item[4.] Differentiation of the objective function:
    %
    \begin{align*}
        &\nabla_{\theta}
        \mathsf{E}
        \left[
            \int_0^T L(X_t,a_t) \dt + \Phi(X_T)  
        \right]
        \\
        &=
        \int_{\mathcal{A}}
        \underbrace{
            \pi_{\theta}(a) \cdot \nabla_{\theta} \log(\pi_{\theta}(a))
        }_{
            \nabla_{\theta} \pi_{\theta}(a)
        }
        \left[
           \Delta t \cdot \sum_{i=0}^{N-1} r_i + \Phi(X_T)
        \right]
        \mathrm{d}a
        \\
        &=
        \mathsf{E}
        \left[
            \sum_{i=0}^{N-1}
            \nabla_{\theta} 
            \log(\pi_{\theta}(a_i))
            \cdot
            C(a_0 , \ldots, a_{N-1})
        \right]
    \end{align*}
    %
    \item[5.] Policy Gradient Theorem: $ C(a_i, \ldots, a_{N-1}) := R_i$
    %
    \begin{align*}
        &\nabla_{\theta} \mathsf{E}
        \left[
            C(\bm{a})
        \right]
        = \mathsf{E}
        \left[
            \sum_{i=0}^{N-1} \nabla_{\theta} 
            \log\left(\pi_{\theta}(a_i)\right)
            \cdot
            R_i
        \right]
        \\
        &\approx
        \frac{1}{M} \sum_{m=1}^M
        \left[
            \sum_{i=0}^{N-1} 
            \nabla_{\theta} 
            \log\left(\pi_{\theta}(a_i^{(m)})\right)
            \cdot
            R_i^{(m)}
        \right]
    \end{align*}
    %
    \item[5.] Gradient method \cite{Plaat_RL} with $M=1$:
    %and $\epsilon_i \sim \mathcal{N}(0, \Delta t)$  
    %
    \begin{align*}
        \theta_{k+1} &= \theta_k - \alpha_k \nabla_{\theta} J(\theta_k)
        \\
        X_{t_{i+1}} 
        &= X_{t_i} + f(X_{t_i},u_i) \Delta t + \sigma Z_i
    \end{align*}
\end{itemize}

\section*{Numerical Example}

We consider the following example with $X_0 = 0$:
%
\begin{align*}
    \min_{u} \int_0^1 X_t +u_t^2  \dt + X_1^2  
    \quad 
    \dX_t = X_t + u_t + 1, 
\end{align*}

and use the stochastic Relaxation:
%
\begin{align*}
    \bm{s}_t =
    \begin{bmatrix}
        t \\
        X_t
    \end{bmatrix}
    \quad
    a_t \sim \mathcal{N}\left(u(\bm{s}_t , \theta), b \right)
\end{align*}

with the parametrization: 
%
\begin{align*}
    u(\bm{s}_t,\theta)
    =
    W_N \cdot 
    \tanh
    \left(
        W_1 \cdot \begin{bmatrix} t \\ X_t \end{bmatrix} + b_1
     \right)
    +
    b_N
\end{align*}

with $W_1 \in \mathbb{R}^{n \times 2}, \bm{b}_1 \in \mathbb{R}^{n}, W_N \in \mathbb{R}^{1 \times n}, b_N \in \mathbb{R}$. 
Where $\theta \in \mathbb{R}^{4n+1}$ and $n$ is a Hyperparameter.
% Hier jetzt konkret die Ableitungen zeigen

For the Reward we compute:
%
\begin{align*}
    r_i = \left(X_{t_i} + a_{t_i}^2\right) \Delta t \quad r_N = X_{t_N}^2
    \Rightarrow
    R_k :=
    \sum_{i=k}^{N} r_i
\end{align*}

For $M=1$ the estimator of our Gradient is given by 
%
\begin{align*}
    \nabla_{\theta} J(\theta)
    \approx
    \hat{g}
    &=
    \sum_{i=0}^{N-1}
    \nabla_{\theta} \log(\pi_{\theta}(a_{t_i} \mid \bm{s}_{t_i}))
    \cdot
    R_{i}
    \\
    &=
    \sum_{i=0}^{N-1}
    \frac{a_{t_i} - u(\bm{s}_{t_i}, \theta)}{b} 
    \cdot R_i
\end{align*}

Using the dynamic for Simulation:
%
\begin{align*}
    X_{t_{i+1}} = X_{t_i} + \Delta t \left(X_{t_i}+ a_{t_i} +1\right) \quad X_{t_0} = 0
\end{align*}
